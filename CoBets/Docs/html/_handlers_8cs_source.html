68 private async Task SendReply(ITelegramBotClient botClient, CancellationToken cancellationToken,
69 long chatId, string data = null)
70 {
71 switch (Stage)
72 {
73 case InputStages.Start:
74 await SendMatchList(botClient, cancellationToken, chatId);
75 break;
76 case InputStages.TeamsInfoOutput:
77 await SendTeamsInfo(botClient, cancellationToken, chatId, data);

114 private async Task SendCoefsInfo(ITelegramBotClient botClient, CancellationToken cancellationToken,
